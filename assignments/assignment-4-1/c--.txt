prog 	: 	func_defn   prog
  	| 	var_decl   prog
  	| 	ε
	;

var_decl 	: 	type   id_list   SEMI
	;

id_list 	: 	ID
  	| 	ID   COMMA   id_list
	;

type 	: 	kwINT
	;

func_defn 	: 	type   ID   LPAREN   opt_formals   RPAREN   LBRACE   opt_var_decls   opt_stmt_list   RBRACE
	;

opt_formals 	: 	ε
  	| 	formals
	;

formals 	: 	type ID   COMMA   formals
  	| 	type ID
	;

opt_var_decls 	: 	ε
  	| 	var_decl   opt_var_decls
	;
